     ```
     const posts = await getCollection('blog');
     const tagCounts = new Map<string, number>();
     posts.forEach(post => {
       post.data.tags.forEach(tag => {
         tagCounts.set(tag, (tagCounts.get(tag) ?? 0) + 1);
       });
     });
     const tags = [...tagCounts.entries()].sort((a, b) => b[1] - a[1]);
     ```
